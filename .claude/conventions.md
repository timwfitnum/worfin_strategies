def calculate_sharpe_ratio(
    returns: pd.Series,
    risk_free_rate: float = 0.04,
    periods: int = 252
) -> float:
    """
    Calculate annualized Sharpe ratio.
    
    Args:
        returns: Daily returns series
        risk_free_rate: Annual risk-free rate
        periods: Number of periods per year
        
    Returns:
        Annualized Sharpe ratio
    """
    excess_returns = returns - risk_free_rate / periods
    return np.sqrt(periods) * excess_returns.mean() / excess_returns.std()

## Async/Await Patterns

### All I/O Operations Must Be Async
```python
import asyncio
import aiohttp
from typing import List, Dict

class DataProvider:
    """Async data provider for market data."""
    
    def __init__(self):
        self.session: Optional[aiohttp.ClientSession] = None
        
    async def __aenter__(self):
        """Async context manager entry."""
        self.session = aiohttp.ClientSession()
        return self
        
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        """Async context manager exit."""
        if self.session:
            await self.session.close()
            
    async def fetch_market_data(self, symbols: List[str]) -> Dict[str, float]:
        """
        Fetch market data for multiple symbols concurrently.
        
        Args:
            symbols: List of symbols to fetch
            
        Returns:
            Dictionary of symbol to price
        """
        tasks = [self._fetch_single(symbol) for symbol in symbols]
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        return {
            symbol: price 
            for symbol, price in zip(symbols, results)
            if not isinstance(price, Exception)
        }
        
    async def _fetch_single(self, symbol: str) -> float:
        """Fetch single symbol price."""
        async with self.session.get(f"/api/quote/{symbol}") as response:
            data = await response.json()
            return data['price']
```

### Async Database Operations
```python
import asyncpg
from contextlib import asynccontextmanager

class DatabaseManager:
    """Async database connection manager."""
    
    def __init__(self, connection_string: str):
        self.connection_string = connection_string
        self.pool: Optional[asyncpg.Pool] = None
        
    async def initialize(self):
        """Initialize connection pool."""
        self.pool = await asyncpg.create_pool(
            self.connection_string,
            min_size=10,
            max_size=20,
            max_queries=50000,
            max_inactive_connection_lifetime=300
        )
        
    @asynccontextmanager
    async def acquire(self):
        """Acquire database connection from pool."""
        async with self.pool.acquire() as connection:
            async with connection.transaction():
                yield connection
                
    async def fetch_positions(self, strategy_id: str) -> List[Position]:
        """Fetch positions for a strategy."""
        async with self.acquire() as conn:
            rows = await conn.fetch(
                """
                SELECT * FROM positions 
                WHERE strategy_id = $1 
                AND status = 'OPEN'
                """,
                strategy_id
            )
            return [Position(**row) for row in rows]
```

## Error Handling

### Comprehensive Error Handling Pattern
```python
import logging
from typing import Optional
from tenacity import retry, stop_after_attempt, wait_exponential

logger = logging.getLogger(__name__)

class BrokerError(Exception):
    """Base class for broker-related errors."""
    pass

class InsufficientMarginError(BrokerError):
    """Raised when account has insufficient margin."""
    pass

class OrderRejectedError(BrokerError):
    """Raised when order is rejected by broker."""
    pass

class OrderExecutor:
    """Handles order execution with comprehensive error handling."""
    
    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=4, max=10),
        reraise=True
    )
    async def execute_order(self, order: Order) -> ExecutionReport:
        """
        Execute order with retry logic and error handling.
        
        Args:
            order: Order to execute
            
        Returns:
            ExecutionReport with status and details
            
        Raises:
            InsufficientMarginError: If margin is insufficient
            OrderRejectedError: If order is rejected
            BrokerError: For other broker-related errors
        """
        try:
            # Pre-execution validation
            await self._validate_order(order)
            
            # Submit to broker
            response = await self.broker_api.submit_order(order)
            
            # Log successful submission
            logger.info(
                f"Order submitted successfully",
                extra={
                    "order_id": order.id,
                    "symbol": order.symbol,
                    "quantity": order.quantity,
                    "response": response
                }
            )
            
            return ExecutionReport(
                order_id=order.id,
                status="SUBMITTED",
                broker_id=response['broker_id']
            )
            
        except asyncio.TimeoutError as e:
            logger.error(
                f"Order submission timeout",
                extra={"order_id": order.id, "error": str(e)}
            )
            # Retry via decorator
            raise
            
        except BrokerAPIError as e:
            if "insufficient margin" in str(e).lower():
                logger.error(
                    f"Insufficient margin for order",
                    extra={"order_id": order.id, "error": str(e)}
                )
                await self.risk_manager.handle_margin_call(order)
                raise InsufficientMarginError(f"Order {order.id}: {e}")
                
            elif "rejected" in str(e).lower():
                logger.error(
                    f"Order rejected by broker",
                    extra={"order_id": order.id, "error": str(e)}
                )
                raise OrderRejectedError(f"Order {order.id}: {e}")
                
            else:
                logger.error(
                    f"Broker API error",
                    extra={"order_id": order.id, "error": str(e)}
                )
                raise BrokerError(f"Order {order.id}: {e}")
                
        except Exception as e:
            # Unexpected error - alert team
            logger.exception(
                f"Unexpected error during order execution",
                extra={"order_id": order.id}
            )
            await self.alert_manager.send_critical_alert(
                f"Order execution failed: {order.id}",
                exception=e
            )
            raise
            
        finally:
            # Always update metrics
            self.metrics.order_attempts.inc()
            
    async def _validate_order(self, order: Order) -> None:
        """Validate order before submission."""
        if order.quantity <= 0:
            raise ValueError(f"Invalid quantity: {order.quantity}")
        
        if order.limit_price and order.limit_price <= 0:
            raise ValueError(f"Invalid limit price: {order.limit_price}")
```

## Logging Standards

### Structured Logging
```python
import structlog
import logging
from typing import Any, Dict

# Configure structured logging
structlog.configure(
    processors=[
        structlog.stdlib.filter_by_level,
        structlog.stdlib.add_logger_name,
        structlog.stdlib.add_log_level,
        structlog.stdlib.PositionalArgumentsFormatter(),
        structlog.processors.TimeStamper(fmt="iso"),
        structlog.processors.StackInfoRenderer(),
        structlog.processors.format_exc_info,
        structlog.processors.UnicodeDecoder(),
        structlog.processors.JSONRenderer()
    ],
    context_class=dict,
    logger_factory=structlog.stdlib.LoggerFactory(),
    cache_logger_on_first_use=True,
)

class StrategyBase:
    """Base strategy with structured logging."""
    
    def __init__(self, strategy_id: str):
        self.strategy_id = strategy_id
        self.logger = structlog.get_logger().bind(
            strategy_id=strategy_id,
            component="strategy"
        )
        
    async def generate_signal(self, data: MarketData) -> Optional[Signal]:
        """Generate trading signal with comprehensive logging."""
        
        # Create request context
        correlation_id = str(uuid.uuid4())
        log = self.logger.bind(
            correlation_id=correlation_id,
            symbol=data.symbol,
            timestamp=data.timestamp.isoformat()
        )
        
        log.debug("Starting signal generation")
        
        try:
            # Calculate signal
            signal_strength = await self._calculate_signal_strength(data)
            log.info(
                "Signal calculated",
                signal_strength=signal_strength,
                bid=data.bid,
                ask=data.ask
            )
            
            if abs(signal_strength) > self.threshold:
                signal = Signal(
                    symbol=data.symbol,
                    direction=int(np.sign(signal_strength)),
                    confidence=abs(signal_strength),
                    strategy_id=self.strategy_id
                )
                
                log.info(
                    "Signal generated",
                    direction=signal.direction,
                    confidence=signal.confidence
                )
                
                return signal
            else:
                log.debug("Signal below threshold", threshold=self.threshold)
                return None
                
        except Exception as e:
            log.exception("Signal generation failed")
            self.metrics.signal_errors.inc()
            raise
```

### Function Documentation
```python
def calculate_kelly_fraction(
    win_probability: float,
    avg_win: float,
    avg_loss: float,
    kelly_multiplier: float = 0.25
) -> float:
    """
    Calculate position size using Kelly Criterion.
    
    Uses the Kelly formula to determine optimal position sizing based on
    expected win rate and win/loss ratio. Applies a fractional Kelly
    multiplier for conservative sizing.
    
    Args:
        win_probability: Probability of winning trade (0-1)
        avg_win: Average winning trade return
        avg_loss: Average losing trade loss (positive number)
        kelly_multiplier: Fraction of Kelly to use (default 0.25)
        
    Returns:
        Optimal position size as fraction of capital
        
    Raises:
        ValueError: If probabilities are outside [0, 1] range
        
    Example:
        >>> size = calculate_kelly_fraction(0.6, 0.02, 0.01, 0.25)
        >>> print(f"Position size: {size:.2%}")
        Position size: 5.00%
        
    Note:
        Using full Kelly (multiplier=1.0) often leads to excessive
        volatility. Most practitioners use 0.25-0.5 Kelly.
    """
    if not 0 <= win_probability <= 1:
        raise ValueError(f"Invalid win probability: {win_probability}")
        
    if avg_loss == 0:
        return 0
        
    # Kelly formula: f = p - q/b
    # where p = win probability, q = loss probability, b = win/loss ratio
    q = 1 - win_probability
    b = avg_win / avg_loss
    
    kelly = win_probability - q / b
    
    # Apply fractional Kelly and ensure non-negative
    return max(0, kelly * kelly_multiplier)
```

## Performance Optimization Rules

### Vectorization Over Loops
```python
# ❌ Bad - Using loops
def calculate_returns(prices: List[float]) -> List[float]:
    returns = []
    for i in range(1, len(prices)):
        returns.append((prices[i] - prices[i-1]) / prices[i-1])
    return returns

# ✅ Good - Using vectorization
def calculate_returns(prices: pd.Series) -> pd.Series:
    return prices.pct_change()
    
# Or with numpy
def calculate_returns(prices: np.ndarray) -> np.ndarray:
    return np.diff(prices) / prices[:-1]
```

### Efficient Pandas Operations
```python
# ❌ Bad - Iterating over DataFrame rows
for index, row in df.iterrows():
    df.at[index, 'signal'] = calculate_signal(row['price'], row['volume'])

# ✅ Good - Using vectorized operations
df['signal'] = calculate_signal(df['price'], df['volume'])

# Or using apply when necessary
df['signal'] = df.apply(
    lambda row: calculate_signal(row['price'], row['volume']),
    axis=1

```
